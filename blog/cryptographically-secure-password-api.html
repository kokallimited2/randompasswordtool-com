Building a Cryptographically Secure Password API in Node.js and Python

4 May 2026 · 9 min read · Alex Rivera, Staff Security Engineer

When password generation needs to be centralised — for provisioning pipelines, on-boarding automation, or multi-service credential management — building a dedicated API makes sense. This walkthrough covers both Node.js (Express) and Python (FastAPI) implementations, with rate limiting, API key validation, and structured output.

Node.js Implementation (Express)

The core generation module uses Node's built-in crypto module — the same OS CSPRNG as the browser Web Crypto API, accessible server-side without additional dependencies.

// generator.js
const { randomFillSync } = require('crypto');

const CHARSETS = {
  u: 'ABCDEFGHIJKLMNOPQRSTUVWXYZ',
  l: 'abcdefghijklmnopqrstuvwxyz',
  n: '0123456789',
  s: '!@#$%^&*()-_=+[]{}|;:,.?',
};
const AMBIGUOUS = new Set(['0','O','l','1','I']);

function buildCharset(flags, excludeAmbiguous) {
  let pool = [...flags].map(f => CHARSETS[f] || '').join('');
  if (excludeAmbiguous)
    pool = [...pool].filter(c => !AMBIGUOUS.has(c)).join('');
  return pool || CHARSETS.u + CHARSETS.l + CHARSETS.n;
}

function generate({ count, length, charset, prefix }) {
  const pool = buildCharset(charset, charset.includes('a'));
  const buf = Buffer.alloc(count * length);
  randomFillSync(buf); // crypto.getRandomValues equivalent
  const passwords = [];
  for (let i = 0; i < count; i++) {
    let pwd = '';
    for (let j = 0; j < length; j++)
      pwd += pool[buf[i * length + j] % pool.length];
    passwords.push(prefix + pwd);
  }
  return passwords;
}

Python Implementation (FastAPI)

Python's secrets module (introduced in 3.6) wraps os.urandom() with a clean API. Never use random.choice() for security purposes.

# generator.py
import secrets, string

CHARSETS = {
    'u': string.ascii_uppercase,
    'l': string.ascii_lowercase,
    'n': string.digits,
    's': '!@#$%^&*()-_=+[]{}|;:,.?',
}
AMBIGUOUS = set('0Ol1I')

def build_charset(flags: str, excl_amb: bool) -> str:
    pool = ''.join(CHARSETS.get(f, '') for f in flags)
    if excl_amb:
        pool = ''.join(c for c in pool if c not in AMBIGUOUS)
    return pool or CHARSETS['u'] + CHARSETS['l'] + CHARSETS['n']

def generate(count, length, charset, prefix) -> list[str]:
    pool = build_charset(charset, 'a' in charset)
    return [
        prefix + ''.join(secrets.choice(pool) for _ in range(length))
        for _ in range(count)
    ]

Rate Limiting

Any credential generation endpoint is a high-value target. Implement rate limiting at the application layer (express-rate-limit / slowapi) and at the infrastructure layer (nginx, API Gateway). Store state in Redis for distributed deployments.

# Python/FastAPI with slowapi
from slowapi import Limiter
from slowapi.util import get_remote_address

limiter = Limiter(key_func=get_remote_address)

@app.get("/api/v1/generate")
@limiter.limit("100/minute;1000/day")
async def generate_passwords(request: Request, ...):
    ...
Production note: Log all generation requests with timestamp, API key hash, count, and length — but never log the generated passwords themselves. Include request IDs for audit trail correlation.
